137  VgaIOSetLine(x+x_t, y-y_t, x_t*-2+1, color);

366 void API_draw_simple_line(int x_1, int y_1, int x_2, int y_2, int color)
367 {
368  /* Initializes private variables */
369  int dx = (int)fabs(x_2-x_1), sx = x_1<x_2 ? 1 : -1;
370  int dy = (int)fabs(y_2-y_1), sy = y_1<y_2 ? 1 : -1;
371  int err = (dx>dy ? dx : -dy)/2, e2;
372 
373  /* Draw the line according to the Bresenham's line Algorithm */
374  while(1){
375  VgaIOSetPixel(x_1,y_1,color);
376  if (x_1==x_2 && y_1==y_2) break;
377  e2 = err;
378  if (e2 >-dx) { err -= dy; x_1 += sx; }
379  if (e2 < dy) { err += dx; y_1 += sy; }
380  }
381 }
